<CFPARAM Name="CurrentPage" Default="1">	<!---	Default the page listing to 1	--->

<!---	Get all products for product listing	--->
<CFQUERY NAME="GetItems" DATASOURCE="#PDS#">
	SELECT * FROM Products
	WHERE Part_Family IN (SELECT Part_Family FROM Classes WHERE Computer_Remarketing = '#Class#')
	Order by Mfr_Model
</CFQUERY>

<!---	CF_PageThru will help us break up the product listing into pages.
	The ADDEDPATH parameter is used to pass the search parameter CLASS
	through the links of the generated page-through navigation.
--->
<CF_PAGETHRU TOTALRECORDS="#GetItems.RecordCount#" CURRENTPAGE="#CurrentPage#"
	TEMPLATEURL="#SCRIPT_NAME#" ADDEDPATH="&Class=#Class#">

<CFOUTPUT>
<HTML>
<BODY BGCOLOR="WHITE">

<h3>Product Listing</h3>

<!---	Output the page-through navigation here.  The HTML for the navigation was
	generated by CF_PageThru and returned in the PT_PageThru variable.
--->
<p>#PT_PageThru#
<p>
<table border>
</CFOUTPUT>



<!---	Show part of the product listing here.  CF_PageThru calculated
	PT_StartRow and PT_EndRow based on the current page of the
	listing that is to be displayed (CurrentPage parameter of CF_PageThru).
--->
<CFLOOP Query="GetItems" StartRow="#PT_StartRow#" EndRow="#PT_EndRow#">
<CFOUTPUT>
<tr>
	<td>#Mfr_Model#</td>
	<td>#Brief_Description#</td>
	<td align="center" valign="middle">
	<a href="detail.cfm?MFRModel=#MFR_Model#&CurrentPage=#CurrentPage#">Add</a>
	</td>
</tr>
</CFOUTPUT>
</CFLOOP>

<CFOUTPUT>
</table>

<!---	Output the page-through navigation again here.	--->
<p>#PT_PageThru#

<!---	Display what range in the product list is being displayed.	--->
<p>Displaying #PT_StartRow#-#PT_EndRow# of <Font color="red"><b>#ProdCount#</b></font> products.
</CFOUTPUT>
